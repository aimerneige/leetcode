# [Merge Sorted Array](https://leetcode.com/problems/merge-sorted-array/description/)

从右侧遍历俩个数组，依次将较大值填入 nums1 末尾即可。

```go
func merge(nums1 []int, m int, nums2 []int, n int) {
	i1 := m - 1
	i2 := n - 1
	i := m + n - 1
	for i1 >= 0 && i2 >= 0 {
		if nums1[i1] > nums2[i2] {
			nums1[i] = nums1[i1]
			i1--
		} else {
			nums1[i] = nums2[i2]
			i2--
		}
		i--
	}
	if i2 >= 0 {
		for ; i >= 0; i-- {
			nums1[i] = nums2[i2]
			i2--
		}
	}

	return
}
```
